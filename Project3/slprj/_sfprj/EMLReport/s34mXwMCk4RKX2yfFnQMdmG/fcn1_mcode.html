1% PURPOSE - Parser to set 
 2% INPUTS
 3%   - Mode: What should the system do in response to the user selection
 4%       - scan: Visit every well with a sticker
 5%       - seek color: Visit every well with a sticker of the given color
 6%       - seek position: Visit only the well with the given sticker ID
 7%   - Color Enum: What color the system should visit
 8%   - Position: Angle in radians to visit
 9%   - Image Data: 2D processed data from webcam
10%       - Col 1: Shape
11%       - Col 2: Color
12%       - Col 3: X Position
13%       - Col 4: Y Position
14% OUTPUTS
15%   - positions_to_visit: Vector of fixed destination positions on [-pi, pi) 
16%       to which the motor arm should point in sequence.
17% NOTES
18%   - If mode = 1 then the function ignores requested position and color.
19%   - If mode = 2 then the function ignores requested position.
20%   - If mode = 3 then the function ignores requested color.
21
22
23function angles_to_visit = fcn(mode, color, position, img_data)
24
25MODE_SCAN = 1;
26MODE_SEEK_COLOR = 2;
27MODE_SEEK_POSITION = 3;
28
29SHAPE_COL = 1;
30COLOR_COL = 2;
31X_COL = 3;
32Y_COL = 4;
33
34if mode == MODE_SCAN
35    rect_coords = img_data(:, X_COL:Y_COL);
36    angles_to_visit = rect_to_polar(rect_coords);
37
38elseif mode == MODE_SEEK_COLOR
39    rect_coords = img_data(img_data(:, COLOR_COL) == color, X_COL:Y_COL);
40    angles_to_visit = rect_to_polar(rect_coords);
41
42elseif mode == MODE_SEEK_POSITION
43    coder.varsize('pos_var_size');
44    pos_var_size = position;
45    angles_to_visit = pos_var_size;
46
47else
48    % FIXME this should give X and Y coords
49    angles_to_visit = rect_to_polar([1,1]);
50end
