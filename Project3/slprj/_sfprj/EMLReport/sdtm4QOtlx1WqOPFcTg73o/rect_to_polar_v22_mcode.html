 2% PURPOSE - Convert rectangular object coordinates to angle for
 3%           motor position
 4% INPUTS
 5%   - X, Y coordinates == center coords of each object
 6% OUTPUTS
 7%   - Vector of angles (radians) ordered ascending
 8% NOTES
 9%   - Theta == 0 is at the east of the map
10%   - Positive angles are in south plane
11%   - Negative angles are in north plane
12%   - Camera origin (0,0) is at NW corner
13%   - Game board origin is at image center
14function Angles_sorted = rect_to_polar_v2( center_coords )
15
16X_COL = 1;
17Y_COL = 2;
18
19% TODO: confirm actual frame dimensions
20IMG_LENGTH_X = 640;
21IMG_HEIGHT_Y = 480;
22
23
24% Get the game board center (Approx. Motor Location) 
25% so we can calculate relative angles
26MotX = IMG_LENGTH_X/2;
27
28MotY = IMG_HEIGHT_Y/2;
29
30numObjects = size(center_coords, 1);
31Angles = zeros(numObjects, 1);
32
33for i = 1:numObjects
34    Xdiff = center_coords(i,X_COL) - MotX;
35    Ydiff = center_coords(i,Y_COL) - MotY;
36    Angles(i) = atan(Ydiff/Xdiff);
37   
38    % Quandrant dependent angle math
39    % Q4
40    if ((Xdiff > 0) && (Ydiff < 0))
41        Angles(i) = Angles(i);
42    % Q3
43    elseif((Xdiff < 0) && (Ydiff < 0))
44        Angles(i) = Angles(i) - pi;
45    % Q2
46    elseif((Xdiff < 0) && (Ydiff > 0))
47        Angles(i) = Angles(i) + pi;
48    elseif((Xdiff > 0) && (Ydiff > 0))
49        % may need to fix this, this is 4th quadrant
50        Angles(i) = Angles(i);
51    % Special case: angle is at -pi
52    elseif((Ydiff == 0) && (Xdiff < 0))
53        Angles(i) = -pi;
54    elseif((Xdiff == 0) && (Ydiff == 0))
55        Angles(i) = 0;
56    end
57end
58
59Angles_sorted = sort(Angles);
60
61
62end
