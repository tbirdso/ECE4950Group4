1% PURPOSE - Parser to set 
 2% INPUTS
 3%   - Mode: What should the system do in response to the user selection
 4%       - scan: Visit every well with a sticker
 5%       - seek color: Visit every well with a sticker of the given color
 6%       - seek position: Visit only the well with the given sticker ID
 7%   - Color Enum: What color the system should visit
 8%   - Position: Angle in radians to visit
 9%   - Image Data: 2D processed data from webcam
10%       - Col 1: Color
11%       - Col 2: X Position
12%       - Col 3: Y Position
13% OUTPUTS
14%   - positions_to_visit: Vector of fixed destination positions on [-pi, pi) 
15%       to which the motor arm should point in sequence.
16% NOTES
17%   - If mode = 1 then the function ignores requested position and color.
18%   - If mode = 2 then the function ignores requested position.
19%   - If mode = 3 then the function ignores requested color.
20
21
22function angles_vector = make_angle_vector(mode, color, position, img_data)
23
24% MATLAB expects a fixed-size output matrix so allow up to 100 angles
25% to be propagated
26MAX_ANGLES_TO_VISIT = 100;
27
28MODE_SCAN = 1;
29MODE_SEEK_COLOR = 2;
30MODE_SEEK_POSITION = 3;
31
32COLOR_COL = 1;
33X_COL = 2;
34Y_COL = 3;
35
36angles_vector = zeros(MAX_ANGLES_TO_VISIT, 1);
37
38if mode == MODE_SCAN
39    rect_coords = img_data(:, X_COL:Y_COL);
40    angles_to_visit = rect_to_polar_v2(rect_coords);
41
42elseif mode == MODE_SEEK_COLOR
43    rect_coords = img_regions(img_data(:, COLOR_COL) == color, X_COL:Y_COL);
44    angles_to_visit = rect_to_polar_v2(rect_coords);
45
46elseif mode == MODE_SEEK_POSITION
47    angles_to_visit = position;
48
49else
50    angles_to_visit = [pi; -pi; pi/2; -pi/2];
51end
52
53% Package variable-length vector into fixed-length vector
54angles_vector(1:size(angles_to_visit,1), :) = angles_to_visit;
